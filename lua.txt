-- Estados
local autoMoneyAtivo = false

local autoIFoodAtivo = false
local autoInfMoneyAtivo = false
local antiRagdollAtivo = false

local noclipConnection = nil

local function getHRP()
    local char = LocalPlayer.Character
    return char and char:FindFirstChild("HumanoidRootPart")
end

local function getHumanoid()
    local char = LocalPlayer.Character
    return char and char:FindFirstChildOfClass("Humanoid")
end

local function teleport(pos)
    local hrp = getHRP()
    if hrp then hrp.CFrame = CFrame.new(pos + Vector3.new(0, 3, 0)) end
end

-- NOCLIP
local function ativarNoclip()
    if noclipConnection then noclipConnection:Disconnect() end
    noclipConnection = RunService.Heartbeat:Connect(function()
        if not noclipAtivo then return end
        local char = LocalPlayer.Character
        if not char then return end
        for _, part in ipairs(char:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end)
end

local function desativarNoclip()
    if noclipConnection then
        noclipConnection:Disconnect()
        noclipConnection = nil
    end
    local char = LocalPlayer.Character
    if char then
        for _, part in ipairs(char:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = true
            end
        end
    end
end

-- AUTO MONEY
task.spawn(function()
    local coletados = {}
    while true do
        task.wait(0.5)
        if not autoMoneyAtivo then continue end
        for obj in pairs(coletados) do
            if not obj.Parent then coletados[obj] = nil end
        end
        for _, obj in ipairs(workspace:GetDescendants()) do
            if not autoMoneyAtivo then break end
            if obj:IsA("Model") and obj.Name == "Money" and not coletados[obj] then
                local root = obj:FindFirstChildWhichIsA("BasePart")
                if root then
                    teleport(root.Position)
                    task.wait(0.1)
                    local prompt = obj:FindFirstChildWhichIsA("ProximityPrompt", true)
                    if prompt then
                        fireproximityprompt(prompt)
                        coletados[obj] = true
                    end
                    task.wait(0.15)
                end
            end
        end
    end
end)

-- AUTO ATM
task.spawn(function()
    local coletados = {}
    while true do
        task.wait(1)
        if not autoATMAtivo then continue end
        for obj in pairs(coletados) do
            if not obj.Parent then coletados[obj] = nil end
        end
        for _, atm in ipairs(workspace:GetChildren()) do
            if not autoATMAtivo then break end
            if atm.Name == "ATM" then
                local prompt = atm:FindFirstChild("PRoubar", true)
                if prompt and prompt.Enabled then
                    local part = atm:FindFirstChildWhichIsA("BasePart")
                    if part then
                        teleport(part.Position)
                        task.wait(0.3)
                        fireproximityprompt(prompt)
                        task.wait(0.5)
                    end
                end
            end
        end
        for _, obj in ipairs(workspace:GetDescendants()) do
            if not autoATMAtivo then break end
            if obj:IsA("Model") and obj.Name == "Money" and not coletados[obj] then
                local root = obj:FindFirstChildWhichIsA("BasePart")
                if root then
                    teleport(root.Position)
                    task.wait(0.1)
                    local prompt = obj:FindFirstChildWhichIsA("ProximityPrompt", true)
                    if prompt then
                        fireproximityprompt(prompt)
                        coletados[obj] = true
                    end
                    task.wait(0.15)
                end
            end
        end
    end
end)

-- AUTO IFOOD
task.spawn(function()
    GPS.OnClientEvent:Connect(function(arg1, arg2)
        if not autoIFoodAtivo then return end
        if typeof(arg1) ~= "Instance" or arg2 ~= true then return end
        teleport(arg1.Position)
        task.wait(0.5)
        local prompt, t = nil, 0
        while not prompt and t < 8 do
            prompt = arg1:FindFirstChild("PEntrega")
            if prompt and not prompt.Enabled then prompt = nil end
            task.wait(0.5)
            t = t + 0.5
        end
        if prompt then
            fireproximityprompt(prompt)
        end
        task.wait(3)
        if autoIFoodAtivo then BuscarPedido:FireServer() end
    end)
    while true do
        task.wait(1)
        if not autoIFoodAtivo then continue end
        BuscarPedido:FireServer()
        task.wait(5)
    end
end)

-- ANTI RAGDOLL
task.spawn(function()
    while true do
        task.wait(0.1)
        if not antiRagdollAtivo then continue end
        local char = LocalPlayer.Character
        if not char then continue end
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then
            if hum:GetState() == Enum.HumanoidStateType.Ragdoll or
               hum:GetState() == Enum.HumanoidStateType.FallingDown then
                hum:ChangeState(Enum.HumanoidStateType.GettingUp)
            end
            hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
            hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
        end
    end
end)

-- PERFORMANCE MODE
local function ativarPerformance()
    settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    Lighting.GlobalShadows = false
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ParticleEmitter") or obj:IsA("Smoke") or
           obj:IsA("Fire") or obj:IsA("Sparkles") then
            obj.Enabled = false
        end
    end
end

local function desativarPerformance()
    settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic
    Lighting.GlobalShadows = true
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ParticleEmitter") or obj:IsA("Smoke") or
           obj:IsA("Fire") or obj:IsA("Sparkles") then
            obj.Enabled = true
        end
    end
end

-- AUTO PRENDER
local function acharPromptPrender(char)
    for _, v in ipairs(char:GetDescendants()) do
        if v:IsA("ProximityPrompt") and
           (v.ActionText:find("Jail") or v.Name:find("Jail") or v.Name:find("Prender")) then
            return v
        end
    end
    return nil
end

RS.Remotes:WaitForChild("Prender").OnClientEvent:Connect(function(player)
    if not autoPrenderAtivo then return end
    if not player or not player.Character then return end
    if presoAtualmente then return end
    presoAtualmente = true

    task.spawn(function()
        local char = player.Character
        local prompt = nil
        local t = 0

        while not prompt and t < 5 do
            prompt = acharPromptPrender(char)
            task.wait(0.2)
            t = t + 0.2
        end

        if not prompt then
            presoAtualmente = false
            return
        end

        local preso = false
        local conn = RunService.Heartbeat:Connect(function()
            if preso then return end
            local myChar = LocalPlayer.Character
            local myHRP = myChar and myChar:FindFirstChild("HumanoidRootPart")
            local targetHRP = char:FindFirstChild("HumanoidRootPart")

            if not myHRP or not targetHRP or not char.Parent then
                preso = true
                return
            end

            myHRP.CFrame = CFrame.new(targetHRP.Position + Vector3.new(0, 2, 0))
        end)

        local tentativas = 0
        while not preso and tentativas < 50 do
            local targetHRP = char:FindFirstChild("HumanoidRootPart")
            if not targetHRP or not char.Parent then break end

            pcall(function() fireproximityprompt(prompt) end)
            task.wait(0.1)
            tentativas = tentativas + 1
        end

        conn:Disconnect()
        presoAtualmente = false
    end)
end)

-- ESP MONEY
local function limparESP()
    for _, label in pairs(espLabels) do
        if label and label.Parent then
            label:Destroy()
        end
    end
    espLabels = {}
end

-- ESP PLAYER
local function limparESPPlayer()
    for _, label in pairs(espPlayerLabels) do
        if label and label.Parent then
            label:Destroy()
        end
    end
    espPlayerLabels = {}
end

-- ESP MONEY RENDER
RunService.RenderStepped:Connect(function()
    if not espMoneyAtivo then
        limparESP()
        return
    end

    local existentes = {}

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Model") and obj.Name == "Money" then
            local root = obj:FindFirstChildWhichIsA("BasePart")
            if root then
                existentes[obj] = true

                if not espLabels[obj] then
                    local bb = Instance.new("BillboardGui")
                    bb.Name = "ESP_Money"
                    bb.Size = UDim2.fromOffset(80, 30)
                    bb.StudsOffset = Vector3.new(0, 3, 0)
                    bb.AlwaysOnTop = true
                    bb.Adornee = root

                    local lbl = Instance.new("TextLabel", bb)
                    lbl.Size = UDim2.fromScale(1, 1)
                    lbl.BackgroundTransparency = 1
                    lbl.Text = "💰 Money"
                    lbl.TextColor3 = Color3.fromRGB(255, 220, 0)
                    lbl.TextStrokeTransparency = 0
                    lbl.TextScaled = true
                    lbl.Font = Enum.Font.GothamBold

                    bb.Parent = root
                    espLabels[obj] = bb
                end

                local hrp = getHRP()
                if hrp and espLabels[obj] then
                    local dist = math.floor((hrp.Position - root.Position).Magnitude)
                    local lbl = espLabels[obj]:FindFirstChildOfClass("TextLabel")
                    if lbl then
                        lbl.Text = "💰 Money\n" .. dist .. "m"
                    end
                end
            end
        end
    end

    for obj, label in pairs(espLabels) do
        if not existentes[obj] then
            if label and label.Parent then label:Destroy() end
            espLabels[obj] = nil
        end
    end
end)

-- ESP PLAYER RENDER
RunService.RenderStepped:Connect(function()
    if not espPlayerAtivo then
        limparESPPlayer()
        return
    end

    local existentes = {}

    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer then continue end
        local char = player.Character
        if not char then continue end
        local root = char:FindFirstChild("HumanoidRootPart")
        if not root then continue end
        
        existentes[player] = true

        if not espPlayerLabels[player] then
            local bb = Instance.new("BillboardGui")
            bb.Name = "ESP_Player"
            bb.Size = UDim2.fromOffset(100, 40)
            bb.StudsOffset = Vector3.new(0, 3, 0)
            bb.AlwaysOnTop = true
            bb.Adornee = root

            local lbl = Instance.new("TextLabel", bb)
            lbl.Size = UDim2.fromScale(1, 1)
            lbl.BackgroundTransparency = 1
            lbl.Text = player.Name
            lbl.TextColor3 = Color3.fromRGB(255, 0, 0)
            lbl.TextStrokeTransparency = 0
            lbl.TextScaled = true
            lbl.Font = Enum.Font.GothamBold

            bb.Parent = root
            espPlayerLabels[player] = bb
        else
            espPlayerLabels[player].Adornee = root
        end
    end

    for player, label in pairs(espPlayerLabels) do
        if not existentes[player] then
            if label and label.Parent then label:Destroy() end
            espPlayerLabels[player] = nil
        end
    end
end)
